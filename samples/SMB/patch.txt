//palette 0f162718

text d57 SORRY BRO       
text d6b SORRY WEEG-MAN  
text d7f PRINCESS AINT HERE BUT
text d98 YA GOT ME!     

if cheats==true
    // Infinite Lives
    hex 11d9 ad

    // world select always enabled
    put 26f eaea
end if
